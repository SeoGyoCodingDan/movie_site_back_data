	function fn_chkDataRange(obj, type) {
		//날짜 형식 및 길이 유효성 체크
		if(!gfn_IsValidateYMD($(obj).val())) return;

		//직접입력한 경우 (yyyy-mm-dd 형식으로 변경)
		if($(obj).val().length == 8){
			$(obj).val($(obj).val().substring(0,4) + "-" + $(obj).val().substring(4,6) + "-" + $(obj).val().substring(6,8));
		}

		if(type == "START"){

			$("#tmpDate").datepicker("setDate",$("#cal_start").datepicker("getDate"));
			$("#tmpDate").datepicker("setDate","c+6d");
			if($("#cal_start").datepicker("getDate") > $("#cal_end").datepicker("getDate")){
				$("#cal_end").datepicker("setDate", $("#tmpDate").datepicker("getDate"));
			}
			/* else if($("#tmpDate").datepicker("getDate") < $("#cal_end").datepicker("getDate") ){
				$("#cal_end").datepicker("setDate", $(obj).datepicker("getDate"));
				$("#cal_end").datepicker("setDate", "c+6d");
			} */
		}
		else{

			$("#tmpDate").datepicker("setDate",$("#cal_end").datepicker("getDate"));
